+from __future__ import annotations
+from typing import Dict, Any
+
+
+class NarrativeBrain:
+    """
+    LLM-powered narrative intelligence layer.
+
+    Interprets world + story state and returns:
+    - narrative intent
+    - suggested events
+    - arc evolution
+    - pacing guidance
+    """
+
+    def __init__(self, llm_client):
+        self.llm = llm_client
+
+    def evaluate(self, context: Dict[str, Any]) -> Dict[str, Any]:
+        prompt = self._build_prompt(context)
+
+        raw = self.llm.complete(prompt)
+
+        return self._parse(raw)
+
+    def _build_prompt(self, context):
+        return f"""
+You are a narrative director AI.
+
+Analyze the current RPG world state and decide:
+
+1. What is the current story about?
+2. What tension direction should occur? (increase / decrease / stabilize)
+3. What meaningful event should happen next?
+4. Should any story arcs progress or resolve?
+
+WORLD:
+{context["world"]}
+
+STORY STATE:
+{context["story"]}
+
+ACTIVE ARCS:
+{context["arcs"]}
+
+Return JSON:
+{{
+  "intent": "...",
+  "tension_delta": float,
+  "events": [],
+  "arc_updates": [],
+  "pacing": "slow|normal|fast"
+}}
+"""
+
+    def _parse(self, raw: str) -> Dict[str, Any]:
+        import json
+        try:
+            return json.loads(raw)
+        except Exception:
+            return {"intent": "fallback", "events": []}

+class NarrativeMemoryAdapter:
+    """
+    Reduces world + story into token-efficient summary.
+    """
+
+    def compress(self, world, story_state, arc_manager):
+        return {
+            "summary": self._summarize(world),
+            "tension": story_state.tension,
+            "phase": story_state.phase,
+            "arcs": arc_manager.get_active_arc_ids()
+        }
+
+    def _summarize(self, world):
+        events = world.get("recent_events", [])[-10:]
+        return [e.get("type") for e in events]
