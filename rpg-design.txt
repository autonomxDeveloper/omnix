class Scene:
    def __init__(self, id, type, location, participants, description, stakes):
        self.id = id
        self.type = type  # "coup", "battle", "trade", "dialogue"
        self.location = location
        self.participants = participants  # character_ids
        self.description = description
        self.stakes = stakes
        self.resolved = False
Scene Engine
class SceneEngine:
    def __init__(self):
        self.active_scenes = []
        self.scene_history = []

    def generate_from_events(self, events):
        scenes = []

        for e in events:
            if e["type"] == "coup":
                scenes.append(self._create_coup_scene(e))

            elif e["type"] == "shortage":
                scenes.append(self._create_crisis_scene(e))

            elif e["type"] == "faction_conflict":
                scenes.append(self._create_battle_scene(e))

        self.active_scenes.extend(scenes)
        return scenes
Example Scene Builder
def _create_coup_scene(self, event):

class Character:

        self.name = name
        self.beliefs = {}  # {"faction_x": -0.5}
        self.goals = []    # ["gain power", "protect city"]

Engine
class CharacterEngine:
    def __init__(self):
        self.characters = {}

    def get_or_create(self, char_id, name=None):
        if char_id not in self.characters:
            self.characters[char_id] = Character(char_id, name or char_id)
        return self.characters[char_id]
Belief Updates from Events
def update_from_events(self, events):
    for e in events:
        if e["type"] == "coup":
            old = self.get_or_create(e["old_leader"])
            new = self.get_or_create(e["new_leader"])

            old.beliefs["power"] = -1.0
            new.beliefs["power"] = 1.0

Structure
class NarrativeMemory:
    def __init__(self, max_entries=100):
        self.events = []
        self.summaries = []
        self.max_entries = max_entries
Add Event
def add_events(self, events):
    for e in events:
        if e.get("importance", 0) > 0.5:
            self.events.append(e)

    if len(self.events) > self.max_entries:
        self._summarize()
Summarization (LLM Hook)
def _summarize(self):
    summary = {
        "type": "summary",
        "text": f"{len(self.events)} major events occurred including coups and wars."
    }

    self.summaries.append(summary)
    self.events = []

Arc Model
class StoryArc:
    def __init__(self, id, setup, payoff_condition):
        self.id = id
        self.setup = setup
        self.payoff_condition = payoff_condition
        self.completed = False
Engine
class StoryArcEngine:
    def __init__(self):
        self.arcs = []

    def register_arc(self, arc):
        self.arcs.append(arc)

    def update(self, world_state):
        completed = []

        for arc in self.arcs:
            if not arc.completed and arc.payoff_condition(world_state):
                arc.completed = True
                completed.append(arc)

        return completed

class NarrativeRenderer:
    def render(self, scenes, memory, characters):
        scene_texts = []

        for s in scenes:
            scene_texts.append(
                f"Scene: {s.type} at {s.location}. Stakes: {s.stakes}."
            )

        summary = " ".join([m["text"] for m in memory.summaries[-3:]])

        return {
            "scene_text": "\n".join(scene_texts),
            "memory_summary": summary
        }
