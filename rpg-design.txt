from typing import List, Dict, Any


class Director:

    def __init__(self, emergence_tracker, event_engine):
        self.emergence_tracker = emergence_tracker
        self.event_engine = event_engine
        self.history = []

    def tick(self, world_state, npcs, recent_outcomes):
        """
        Called once per game tick AFTER all NPCs act
        """

        signals = self.emergence_tracker.analyze(world_state, npcs, recent_outcomes)

        decision = self._decide_intervention(signals)

        if decision:
            event = self.event_engine.create_event(decision, world_state)
            self.event_engine.apply_event(event, world_state)
            self.history.append(event)

        return decision

    def _decide_intervention(self, signals: Dict[str, Any]):

        # 1. Stagnation → inject twist
        if signals.get("stagnation", 0) > 0.7:
            return {"type": "twist", "intensity": 0.6}

        # 2. Conflict spike → escalate
        if signals.get("conflict", 0) > 0.8:
            return {"type": "escalation", "intensity": 0.9}

        # 3. Repeated failures → assist or punish
        if signals.get("failure_spike", 0) > 0.6:
            return {"type": "intervention", "mode": "assist"}

        # 4. High divergence → chaos
        if signals.get("divergence", 0) > 0.75:
            return {"type": "chaos", "intensity": 0.8}

        return None

import random


class EventEngine:

    def create_event(self, decision, world_state):

        event_type = decision["type"]

        if event_type == "twist":
            return {
                "name": "unexpected_betrayal",
                "description": "An ally turns against the group",
            }

        if event_type == "escalation":
            return {
                "name": "enemy_reinforcements",
                "description": "More enemies arrive",
            }

        if event_type == "intervention":
            return {
                "name": "supply_drop",
                "description": "Helpful resources appear nearby",
            }

        if event_type == "chaos":
            return {
                "name": "environmental_hazard",
                "description": "The environment becomes dangerous",
            }

        return None

    def apply_event(self, event, world_state):

        if not event:
            return

        name = event["name"]

        if name == "enemy_reinforcements":
            world_state["enemy_count"] += 2

        elif name == "supply_drop":
            world_state["resources"] += 3

        elif name == "environmental_hazard":
            world_state["danger_level"] += 2

        elif name == "unexpected_betrayal":
            world_state["trust_level"] -= 0.5

class EmergenceAdapter:

    def analyze(self, world_state, npcs, outcomes):

        return {
            "stagnation": self._stagnation(npcs),
            "conflict": self._conflict(world_state),
            "failure_spike": self._failure(outcomes),
            "divergence": self._divergence(world_state),
        }

    def _stagnation(self, npcs):
        idle_count = sum(1 for n in npcs if n.last_action == "idle")
        return idle_count / max(len(npcs), 1)

    def _conflict(self, world_state):
        return world_state.get("enemy_count", 0) / 10

    def _failure(self, outcomes):
        if not outcomes:
            return 0
        failures = sum(1 for o in outcomes if not o.success)
        return failures / len(outcomes)

    def _divergence(self, world_state):
        return world_state.get("chaos", 0.0)
