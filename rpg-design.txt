class StoryDirector:
    def __init__(self):
        self.phase = "intro"   # intro → build → tension → climax → resolution
        self.tension = 0.0     # 0 → 1
        self.arc = None

        self.history = []
        self.cooldowns = {}

    # -------------------------
    # MAIN ENTRY POINT
    # -------------------------
    def adjust_goal(self, npc, proposed_goal, context):
        self._update_tension(context)
        self._update_phase()

        goal = proposed_goal

        # Apply arc shaping
        if self.arc == "conflict":
            goal = self._bias_conflict(npc, goal)

        elif self.arc == "alliance":
            goal = self._bias_alliance(npc, goal)

        elif self.arc == "mystery":
            goal = self._bias_mystery(npc, goal)

        # Apply pacing rules
        goal = self._apply_pacing(npc, goal)

        return goal
3. 🎯 Tension System
def _update_tension(self, context):
    events = context.get("recent_events", [])

    delta = 0.0

    for e in events:
        if e["type"] == "damage":
            delta += 0.05
        elif e["type"] == "death":
            delta += 0.2
        elif e["type"] == "assist":
            delta -= 0.03

    self.tension = max(0.0, min(1.0, self.tension + delta))
4. 🎬 Phase Progression
def _update_phase(self):
    if self.tension < 0.2:
        self.phase = "intro"
    elif self.tension < 0.5:
        self.phase = "build"
    elif self.tension < 0.8:
        self.phase = "tension"
    elif self.tension >= 0.8:
        self.phase = "climax"
5. ⚖️ Goal Shaping (CORE MAGIC)
Conflict Bias
def _bias_conflict(self, npc, goal):
    if "attack" in goal["name"]:
        goal["priority"] *= 1.3

    if "talk" in goal["name"]:
        goal["priority"] *= 0.7

    return goal
Alliance Bias
def _bias_alliance(self, npc, goal):
    if "assist" in goal["name"]:
        goal["priority"] *= 1.5

    if "attack" in goal["name"]:
        goal["priority"] *= 0.6

    return goal
Mystery Bias
def _bias_mystery(self, npc, goal):
    if "explore" in goal["name"]:
        goal["priority"] *= 1.4

    if "attack" in goal["name"]:
        goal["priority"] *= 0.8

    return goal
6. ⏱️ Pacing Control (VERY IMPORTANT)
def _apply_pacing(self, npc, goal):
    if self.phase == "intro":
        if "attack" in goal["name"]:
            goal["priority"] *= 0.3

    elif self.phase == "tension":
        goal["priority"] *= 1.2

    elif self.phase == "climax":
        if "attack" in goal["name"]:
            goal["priority"] *= 1.5

    return goal

11. 🚨 Anti-Repetition Guard (VERY IMPORTANT)
def prevent_repetition(self, npc, goal):
    key = (npc.id, goal["name"])

    if self.cooldowns.get(key, 0) > 0:
        goal["priority"] *= 0.2

    self.cooldowns[key] = 3
    return goal
