import hashlib
import json

@staticmethod
def compute_event_id(event) -> str:
    data = {
        "type": event.type,
        "payload": event.payload,
        "parent": event.parent_id,
        "tick": event.tick,
    }

    serialized = json.dumps(data, sort_keys=True)
    return hashlib.sha256(serialized.encode()).hexdigest()[:16]
