+from collections import defaultdict
+
+
+class ReputationGraph:
+    """
+    Directed weighted graph:
+    A -> B = how A feels about B
+    """
+
+    def __init__(self):
+        self.edges = defaultdict(lambda: defaultdict(float))
+
+    def get(self, a, b):
+        return self.edges[a][b]
+
+    def update(self, a, b, delta):
+        val = self.edges[a][b] + delta
+        self.edges[a][b] = max(-1.0, min(1.0, val))
+
+    def neighbors(self, a):
+        return self.edges[a]
+
+    def top_relations(self, a, n=5):
+        return sorted(
+            self.edges[a].items(),
+            key=lambda x: x[1],
+            reverse=True
+        )[:n]
2️⃣ ALLIANCE SYSTEM
+++ src/app/rpg/ai/social/alliance_system.py
+class AllianceSystem:
+
+    def __init__(self):
+        self.factions = {}  # faction_id -> set(npc_ids)
+
+    def form_alliance(self, a, b):
+        for fid, members in self.factions.items():
+            if a in members or b in members:
+                members.update([a, b])
+                return fid
+
+        fid = f"faction_{len(self.factions)}"
+        self.factions[fid] = {a, b}
+        return fid
+
+    def break_alliance(self, a, b):
+        for fid, members in self.factions.items():
+            if a in members and b in members:
+                members.discard(a)
+                members.discard(b)
+
+    def get_faction(self, npc_id):
+        for fid, members in self.factions.items():
+            if npc_id in members:
+                return fid
+        return None
3️⃣ RUMOR SYSTEM (INFORMATION SPREAD)
+++ src/app/rpg/ai/social/rumor_system.py
+import random
+
+
+class RumorSystem:
+
+    def __init__(self):
+        self.rumors = []  # [{content, source, ttl}]
+
+    def add_rumor(self, content, source):
+        self.rumors.append({
+            "content": content,
+            "source": source,
+            "ttl": 5
+        })
+
+    def tick(self, npcs):
+        new_rumors = []
+
+        for rumor in self.rumors:
+            rumor["ttl"] -= 1
+            if rumor["ttl"] <= 0:
+                continue
+
+            for npc in npcs:
+                if random.random() < 0.3:
+                    new_rumors.append({
+                        "content": rumor["content"],
+                        "source": npc,
+                        "ttl": rumor["ttl"]
+                    })
+
+        self.rumors.extend(new_rumors)
4️⃣ BETRAYAL PROPAGATION
+++ src/app/rpg/ai/social/social_engine.py
+class SocialEngine:
+
+    def __init__(self, reputation_graph, alliance_system, rumor_system):
+        self.rep = reputation_graph
+        self.alliances = alliance_system
+        self.rumors = rumor_system
+
+    def process_event(self, event):
+        t = event.get("type")
+        a = event.get("actor")
+        b = event.get("target")
+
+        if not a or not b:
+            return
+
+        if t == "help":
+            self.rep.update(a, b, 0.2)
+            self.rep.update(b, a, 0.2)
+
+        elif t == "attack":
+            self.rep.update(a, b, -0.5)
+            self.rep.update(b, a, -0.7)
+
+            # propagate to allies
+            fid = self.alliances.get_faction(b)
+            if fid:
+                for ally in self.alliances.factions[fid]:
+                    self.rep.update(ally, a, -0.3)
+
+            self.rumors.add_rumor(f"{a} attacked {b}", a)
+
+        elif t == "betrayal":
+            self.rep.update(a, b, -1.0)
+            self.alliances.break_alliance(a, b)
+            self.rumors.add_rumor(f"{a} betrayed {b}", a)
5️⃣ GROUP DECISION ENGINE
+++ src/app/rpg/ai/social/group_decision.py
+from collections import Counter
+
+
+class GroupDecisionEngine:
+
+    def decide(self, npc_ids, decisions):
+        """
+        decisions: {npc_id: NPCDecision}
+        """
+
+        intents = [d.intent for d in decisions.values()]
+        common = Counter(intents).most_common(1)
+
+        if not common:
+            return None
+
+        dominant_intent = common[0][0]
+
+        # choose leader action
+        for npc_id, d in decisions.items():
+            if d.intent == dominant_intent:
+                return d
+
+        return None
