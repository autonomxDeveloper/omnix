from typing import List, Dict, Any
from .narrative_event import NarrativeEvent
import uuid


class NarrativeDirector:

    def __init__(self):
        self.recent_events: List[NarrativeEvent] = []

    def convert_events(self, world_events: List[Dict[str, Any]]) -> List[NarrativeEvent]:
        """Convert raw world events into narrative events."""
        output = []

        for e in world_events:
            ne = NarrativeEvent(
                id=str(uuid.uuid4()),
                type=e.get("type", "unknown"),
                description=e.get("description", ""),
                actors=e.get("actors", []),
                location=e.get("location"),
                importance=self._score_importance(e),
                emotional_weight=self._score_emotion(e),
                tags=e.get("tags", []),
                raw_event=e,
            )
            output.append(ne)

        return output

    def _score_importance(self, event: Dict[str, Any]) -> float:
        base = 0.5

        if event.get("type") in ["combat", "death"]:
            base += 0.4
        if "player" in event.get("actors", []):
            base += 0.3

        return min(base, 1.0)

    def _score_emotion(self, event: Dict[str, Any]) -> float:
        if event.get("type") == "death":
            return 1.0
        if event.get("type") == "combat":
            return 0.7
        return 0.2

    def select_focus_events(self, events: List[NarrativeEvent], max_events: int = 5) -> List[NarrativeEvent]:
        """Pick most important events for narration."""
        events.sort(key=lambda e: (e.importance + e.emotional_weight), reverse=True)
        return events[:max_events]

from typing import List, Dict
from dataclasses import dataclass, field


@dataclass
class Scene:
    id: str
    location: str
    participants: List[str]
    recent_events: List[Dict] = field(default_factory=list)
    mood: str = "neutral"


class SceneManager:

    def __init__(self):
        self.active_scene: Scene | None = None

    def update_scene(self, events: List[Dict]):
        if not events:
            return

        primary = events[0]

        if not self.active_scene:
            self.active_scene = Scene(
                id="scene_1",
                location=primary.get("location", "unknown"),
                participants=primary.get("actors", []),
            )

        self.active_scene.recent_events.extend(events)

        # Keep memory small
        self.active_scene.recent_events = self.active_scene.recent_events[-10:]

    def get_scene_context(self) -> Dict:
        if not self.active_scene:
            return {}

        return {
            "location": self.active_scene.location,
            "participants": self.active_scene.participants,
            "recent_events": self.active_scene.recent_events,
        }

from typing import List
from .narrative_event import NarrativeEvent


class NarrativeGenerator:

    def __init__(self, llm):
        self.llm = llm

    def generate(self, events: List[NarrativeEvent], scene_context: dict) -> str:
        """Turn structured events into narrative text."""

        event_descriptions = "\n".join(
            f"- {e.description}" for e in events
        )

        prompt = f"""
You are a cinematic RPG narrator.

Scene:
Location: {scene_context.get("location")}
Participants: {scene_context.get("participants")}

Events:
{event_descriptions}

Write a vivid, immersive narration of what happens.
Focus on clarity, flow, and emotional impact.
"""

        return self.llm.generate(prompt)

class PlayerLoop:

    def __init__(self, world, director, scene_manager, narrator):
        self.world = world
        self.director = director
        self.scene_manager = scene_manager
        self.narrator = narrator

    def step(self, player_input: str) -> dict:
        """Main game loop step."""

        # 1. Inject player action into world
        player_event = self._convert_input(player_input)
        self.world.inject_event(player_event)

        # 2. Run world simulation tick
        world_events = self.world.world_tick()

        # 3. Convert to narrative events
        narrative_events = self.director.convert_events(world_events)

        # 4. Select important ones
        focus_events = self.director.select_focus_events(narrative_events)

        # 5. Update scene
        self.scene_manager.update_scene([e.raw_event for e in focus_events])

        # 6. Generate narration
        scene_context = self.scene_manager.get_scene_context()
        narration = self.narrator.generate(focus_events, scene_context)

        return {
            "narration": narration,
            "events": focus_events,
        }

    def _convert_input(self, text: str) -> dict:
        return {
            "type": "player_action",
            "description": text,
            "actors": ["player"],
        }
